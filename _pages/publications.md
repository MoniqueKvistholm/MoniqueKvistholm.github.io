<script>
  document.addEventListener("DOMContentLoaded", function() {
    const container = document.getElementById('publicationsContainer');
    const publications = Array.from(container.getElementsByClassName('publication-box'));

    // Sort publications based on the date attribute
    publications.sort((a, b) => {
      const dateA = new Date(a.dataset.date);
      const dateB = new Date(b.dataset.date);
      return dateB - dateA; // Sort in descending order
    });

    // Append sorted publications back to the container
    publications.forEach(publication => {
      container.appendChild(publication);
    });
  });

  function filterPublications(category) {
    const publications = document.querySelectorAll('.publication-box');
    publications.forEach(pub => {
      if (category === 'all' || pub.classList.contains(category)) {
        pub.style.display = 'block';
      } else {
        pub.style.display = 'none';
      }
    });
  }
</script>
